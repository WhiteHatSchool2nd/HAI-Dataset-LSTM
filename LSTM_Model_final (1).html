정규화¶

남은 공정 변수 중 가장 큰 값과 가장 작은 값을 구한 후 각 컬럼별로 Min-Max 정규화를 진행하였다.

In [ ]:
def normalize(df, TAG_MIN, TAG_MAX):
    ndf = df.copy()
    for c in df.columns:
        if TAG_MIN[c] == TAG_MAX[c]:
            ndf[c] = df[c] - TAG_MIN[c]
        else:
            ndf[c] = (df[c] - TAG_MIN[c]) / (TAG_MAX[c] - TAG_MIN[c])
    return ndf

# Min-Max Normalize
TRAIN_DF = normalize(TRAIN_DF_RAW[VALID_COLUMNS_IN_TRAIN_DATASET], TAG_MIN, TAG_MAX).ewm(alpha=0.9).mean()

def sliding_window_unsupervised(df_train, df_target, window_size, feature_columns, answer_column, rolling_size= 10):
    data = df_train[feature_columns].values
    data_test = df_target[feature_columns].values
    ma_values = df_target[feature_columns].rolling(rolling_size, min_periods=1).mean().values
    answers = answer_column.values

    num_samples = len(df_train)
    features = np.empty((num_samples, window_size, len(feature_columns)), dtype=np.float32)
    tests = np.empty((num_samples, window_size, len(feature_columns)), dtype=np.float32)
    train_targets = np.empty((num_samples, 1, len(feature_columns)), dtype=np.float32)
    targets = np.empty((num_samples, 1, len(feature_columns)), dtype=np.float32)
    answer_targets = np.empty(num_samples, dtype=int)
    ma_values = np.empty((num_samples, len(feature_columns)), dtype=np.float32)

    for i in range(num_samples - window_size):
        features[i] = data[i:i+window_size]
        tests[i] = data_test[i:i+window_size]
        train_targets[i] = data[i+window_size]
        targets[i] = data_test[i+window_size]
        answer_targets[i] = 1 if np.any(answers[i:i+window_size] == 1) else 0

    return features, targets, answer_targets, train_targets, tests, ma_values

optimizer = Adam()

def build_autoencoder(window_size, num_features):
    model = Sequential()
    # 인코더
    model.add(LSTM(64, activation='relu', input_shape=(window_size, num_features), return_sequences=False))

    model.add(Dense(64))  # 잠재 공간 표현
    model.add(RepeatVector(1))  # 디코더로 전달할 시퀀스 재생성
    model.add(Dense(64))
    
    # 디코더
    model.add(LSTM(64, activation='relu', input_shape=(window_size+1, num_features), return_sequences=True))
    model.add(TimeDistributed(Dense(len(feature_columns))))

    model.compile(optimizer=optimizer, loss='mse')
    return model

def check_anomaly(predicted_table, target_table, threshold):
    predicted_table[target_table >= threshold] = 1
    return predicted_table

def calculate_table(predicted, targets):
    differences = np.abs(predicted - targets)
    # 이동평균 값
    pt = np.mean(differences, axis=2).flatten()
    ma_values = np.mean(ma, axis=1) 
    as_t = (ma_values + pt) / 2
    return as_t
